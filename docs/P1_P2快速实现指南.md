## P1-5: 图片/视频预览优化

### MediaViewer 组件
```typescript
import { useState } from 'react';
import { X, Download, ChevronLeft, ChevronRight } from 'lucide-react';

interface MediaViewerProps {
  media: { url: string; type: 'image' | 'video' }[];
  initialIndex: number;
  onClose: () => void;
}

export default function MediaViewer({ media, initialIndex, onClose }: MediaViewerProps) {
  const [currentIndex, setCurrentIndex] = useState(initialIndex);
  const current = media[currentIndex];

  const handlePrev = () => {
    setCurrentIndex(i => (i > 0 ? i - 1 : media.length - 1));
  };

  const handleNext = () => {
    setCurrentIndex(i => (i < media.length - 1 ? i + 1 : 0));
  };

  return (
    <div className="fixed inset-0 bg-black z-50 flex items-center justify-center">
      <button onClick={onClose} className="absolute top-4 right-4 text-white">
        <X className="w-8 h-8" />
      </button>

      {media.length > 1 && (
        <>
          <button onClick={handlePrev} className="absolute left-4 text-white">
            <ChevronLeft className="w-8 h-8" />
          </button>
          <button onClick={handleNext} className="absolute right-4 text-white">
            <ChevronRight className="w-8 h-8" />
          </button>
        </>
      )}

      <div className="max-w-[90vw] max-h-[90vh]">
        {current.type === 'image' ? (
          <img src={current.url} className="max-w-full max-h-full object-contain" />
        ) : (
          <video src={current.url} controls className="max-w-full max-h-full" />
        )}
      </div>

      <div className="absolute bottom-4 left-1/2 -translate-x-1/2 text-white">
        {currentIndex + 1} / {media.length}
      </div>
    </div>
  );
}
```

```typescript
// ThemeProvider.tsx
import { createContext, useContext, useEffect, useState } from 'react';

const ThemeContext = createContext<{
  isDark: boolean;
  toggleTheme: () => void;
}>({ isDark: false, toggleTheme: () => {} });

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [isDark, setIsDark] = useState(() => {
    return localStorage.getItem('darkMode') === 'true';
  });

  useEffect(() => {
    if (isDark) {
      document.documentElement.classList.add('dark');
    } else {
      document.documentElement.classList.remove('dark');
    }
    localStorage.setItem('darkMode', isDark.toString());
  }, [isDark]);

  const toggleTheme = () => setIsDark(!isDark);

  return (
    <ThemeContext.Provider value={{ isDark, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export const useTheme = () => useContext(ThemeContext);
```

## P2-3: 聊天记录导出

### 前端实现
```typescript
const exportChatHistory = (messages: Message[], format: 'txt' | 'json' | 'html') => {
  let content = '';
  let mimeType = '';
  let extension = '';

  if (format === 'txt') {
    content = messages.map(m =>
      `[${new Date(m.createdAt).toLocaleString()}] ${m.senderName}: ${m.content}`
    ).join('\n');
    mimeType = 'text/plain';
    extension = 'txt';
  } else if (format === 'json') {
    content = JSON.stringify(messages, null, 2);
    mimeType = 'application/json';
    extension = 'json';
  } else if (format === 'html') {
    content = `
      <!DOCTYPE html>
      <html>
      <head><meta charset="utf-8"><title>聊天记录</title></head>
      <body>
        <h1>聊天记录</h1>
        ${messages.map(m => `
          <div style="margin: 10px 0;">
            <strong>${m.senderName}</strong>
            <span style="color: gray; font-size: 12px;">${new Date(m.createdAt).toLocaleString()}</span>
            <p>${m.content}</p>
          </div>
        `).join('')}
      </body>
      </html>
    `;
    mimeType = 'text/html';
    extension = 'html';
  }

  const blob = new Blob([content], { type: mimeType });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `chat_${Date.now()}.${extension}`;
  a.click();
  URL.revokeObjectURL(url);
};
```
